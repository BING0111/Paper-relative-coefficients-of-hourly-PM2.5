model { 

  # likelihood
  for (i in 1:N) {
    y[i]      ~ dpois(mu[i])
    mu[i]    <- exp(theta[i])
    theta[i] <- inprod(want, x1[hmat[i,]])
  }
  
  want <- beta1 * gamma
  
  # prior
  beta1 ~ dnorm(mu.beta1, tau.beta1)
  mu.beta1 <- 0.05
  tau.beta1 <- pow(sigma.beta1, -2)
  sigma.beta1 <- 0.5
  
  for (g in 1:K) {
	gamma[g] <- gb[g] - mean(gb[]) + 1  # ref: The BUGS Book Page 213
	gb[g] ~ dunif(0,1)
  }
}
